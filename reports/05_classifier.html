1 Complete Workflow

Note

For up- and down-sampling, we employed adaptive balancing: We employed a target-ratio parameter (tunable range: 1 to observed max:min ratio) that governed the degree of class rebalancing. The algorithm computed a reference count from class frequencies (median for <= 10 classes; 60th percentile for 11-20 classes; 65th percentile for > 20 classes), then downsampled classes exceeding reference\(\times\)target ratio using NearMiss and upsampled classes below reference < target ratio using Borderline SMOTE, and regular SMOTE in case of non sufficient danger observations.

Note

Random forests were tuned with 500 trees per ensemble during tuning. The final model on the whole tranining set was trained with 1000 trees.

Note

Variables starting with “dtb.” refer to artist genres from Spotify. We assume that artist genres are assigned with a strategic aim for distribution on the platform (Valverde, 2025). In that regard, they are different from the musical genres assigned by the listeners. The variables are flags indicating whether an artist genre is assigned to a track’s artist or not. To reduce the dimensionality, we used the same hiearchical-tree-based approach as for the MusicBrainz meta-genres.

load_learner_f1_cv <- function(learner, resolution, hash, model_dir, param_names) {
  tuning <- readRDS(
    sprintf(
      "../models/classifier/%s/%s_%s_%s_tuning_history.rds",
      model_dir, learner, resolution, hash
    )
  )
  settings <- readRDS(
    sprintf(
      "../models/classifier/%s/%s_%s_%s_settings.rds",
      model_dir, learner, resolution, hash
    )
  )
  params <- unlist(param_names)
  chosen_params <- setNames(unlist(settings[params]), params)
  # exchange "alpha" with "mixture" if in names for GLMNET
  if (learner == "GLMNET"){
    names(chosen_params) <- gsub("alpha", "mixture", names(chosen_params))
  }
   
  prepare_tuning_topmodels(tuning, chosen_params, top_n = 100) |>
    dplyr::filter(is_chosen) |>
    dplyr::select(
      macro_f1_with_zeros_mean,
      macro_f1_with_zeros_std_err
    ) |>
    dplyr::mutate(learner = learner, detail = resolution)
}

load_resolution_data <- function(resolution_hash) {
  base_path <- sprintf(
    "../models/classifier/lightgbm/lightgbm_%sres_%s",
    names(resolutions)[which(sapply(resolutions, `[[`, "hash") == resolution_hash)],
    resolution_hash
  )
  
  list(
    eval = readRDS(sprintf("%s_evaluation.rds", base_path)),
    settings = readRDS(sprintf("%s_settings.rds", base_path)),
    tuning = readRDS(sprintf("%s_tuning_history.rds", base_path))
  )
}

extract_chosen_params <- function(settings) {
  c(
    tree_depth = settings$tree_depth,
    trees = settings$trees,
    learn_rate = settings$learn_rate,
    mtry = settings$mtry,
    min_n = settings$min_n,
    target_ratio = settings$target_ratio
  )
}
